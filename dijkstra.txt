#include<stdio.h>
#define INF 999
int main(){
 int n,cost[10][10],dist[10],visited[10]={0},i,j,start,min,next;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 for(j=0;j<n;j++) scanf("%d",&cost[i][j]);
 scanf("%d",&start);
 for(i=0;i<n;i++) dist[i]=cost[start][i];
 visited[start]=1;
 for(i=1;i<n;i++){
 min=INF;
 for(j=0;j<n;j++)
 if(!visited[j] && dist[j]<min){
 min=dist[j]; next=j;
 }
 visited[next]=1;
 for(j=0;j<n;j++)
 if(!visited[j] && min+cost[next][j]<dist[j])
 dist[j]=min+cost[next][j];
 }
 for(i=0;i<n;i++) printf("%d -> %d\n",i,dist[i]);
}
