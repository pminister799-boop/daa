#include<stdio.h>
int max(int a,int b){ return a>b?a:b; }
int main(){
 int w[10],p[10],k[10][10],n,m,i,j;
 scanf("%d",&n);
 for(i=1;i<=n;i++) scanf("%d%d",&w[i],&p[i]);
 scanf("%d",&m);
 for(i=0;i<=n;i++){
 for(j=0;j<=m;j++){
 if(i==0 || j==0) k[i][j]=0;
 else if(w[i]<=j)
 k[i][j]=max(p[i]+k[i-1][j-w[i]],k[i-1][j]);
 else
 k[i][j]=k[i-1][j];
 }
 }
 printf("Profit=%d",k[n][m]);
}
