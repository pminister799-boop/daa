#include<stdio.h>
int main(){
 int n,a[10][10],indeg[10]={0},i,j,count=0;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 for(j=0;j<n;j++){
 scanf("%d",&a[i][j]);
 indeg[j]+=a[i][j];
 }
 while(count<n){
 for(i=0;i<n;i++){
 if(indeg[i]==0){
 printf("%d ",i);
 indeg[i]=-1;
 for(j=0;j<n;j++)
 if(a[i][j]) indeg[j]--;
 count++;
 }
 }
 }
}
