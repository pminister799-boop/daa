#include<stdio.h>
int parent[10];
int find(int i){ while(parent[i]) i=parent[i]; return i; }
int uni(int i,int j){ if(i!=j){ parent[j]=i; return 1;} return 0; }
int main(){
 int n,i,j,a,b,u,v,min,mincost=0,cost[10][10],edges=1;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 for(j=0;j<n;j++) scanf("%d",&cost[i][j]);
 while(edges<n){
 min=999;
 for(i=0;i<n;i++)
 for(j=0;j<n;j++)
 if(cost[i][j]<min){
 min=cost[i][j]; a=u=i; b=v=j;
 }
 u=find(u); v=find(v);
 if(uni(u,v)){
 printf("%d-%d=%d\n",a,b,min);
 mincost+=min; edges++;
 }
 cost[a][b]=cost[b][a]=999;
 }
 printf("Cost=%d",mincost);
}
