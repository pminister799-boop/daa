#include<stdio.h>
struct item{float w,p,r;};
int main(){
 struct item a[10],temp;
 int n,i,j;
 float cap,profit=0;
 scanf("%d",&n);
 for(i=0;i<n;i++){
 scanf("%f%f",&a[i].w,&a[i].p);
 a[i].r=a[i].p/a[i].w;
}
 scanf("%f",&cap);
 for(i=0;i<n-1;i++)
 for(j=i+1;j<n;j++)
 if(a[i].r<a[j].r){
 temp=a[i]; a[i]=a[j]; a[j]=temp;
 }
 for(i=0;i<n;i++){
 if(cap>=a[i].w){
 profit+=a[i].p;
 cap-=a[i].w;
 } else {
 profit+=a[i].r*cap;
 break;
 }
 }
 printf("Profit=%f",profit);
}
